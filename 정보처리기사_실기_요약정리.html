1.4 클래스와 객체

// 클래스 정의
public class Person {
    // 필드 (멤버 변수)
    private String name;
    private int age;
    
    // 생성자
    public Person(String name, int age) {
        this.name = name;
        this.age = age;
    }
    
    // 메서드
    public String getName() {
        return name;
    }
    
    public void setName(String name) {
        this.name = name;
    }
    
    // 메서드 오버로딩
    public void print() {
        System.out.println(name + ", " + age);
    }
    
    public void print(String prefix) {
        System.out.println(prefix + name + ", " + age);
    }
}

// 객체 생성 및 사용

2. 데이터베이스 (SQL)

2.1 기본 조회 (SELECT)

-- 기본 조회
SELECT * FROM 테이블명;
SELECT 컬럼1, 컬럼2 FROM 테이블명;

-- 조건 (WHERE)
SELECT * FROM 테이블명 WHERE 조건;
SELECT * FROM 학생 WHERE 나이 > 20;

-- 정렬 (ORDER BY)
SELECT * FROM 테이블명 ORDER BY 컬럼 ASC;  -- 오름차순
SELECT * FROM 테이블명 ORDER BY 컬럼 DESC; -- 내림차순

-- LIMIT
SELECT * FROM 테이블명 LIMIT 10; -- 상위 10개

2.2 집계 함수

SELECT COUNT(*) FROM 테이블명;      -- 행 개수
SELECT SUM(컬럼) FROM 테이블명;      -- 합계
SELECT AVG(컬럼) FROM 테이블명;      -- 평균
SELECT MAX(컬럼) FROM 테이블명;      -- 최대값
SELECT MIN(컬럼) FROM 테이블명;      -- 최소값

-- GROUP BY
SELECT 부서, AVG(급여) 
FROM 직원 
GROUP BY 부서;

-- HAVING (GROUP BY 결과 필터링)
SELECT 부서, AVG(급여) 
FROM 직원 
GROUP BY 부서 
HAVING AVG(급여) > 3000;

2.4 서브쿼리

-- 단일 행 서브쿼리
SELECT * 
FROM 테이블명 
WHERE 컬럼 = (SELECT 컬럼 FROM 테이블2 WHERE 조건);

-- 다중 행 서브쿼리 (IN)
SELECT * 
FROM 테이블명 
WHERE 컬럼 IN (SELECT 컬럼 FROM 테이블2 WHERE 조건);

-- EXISTS
SELECT * 
FROM 테이블1 t1
WHERE EXISTS (SELECT 1 FROM 테이블2 t2 WHERE t2.키 = t1.키);

3. 알고리즘

3.1 정렬

// 배열 정렬
int[] arr = {5, 2, 8, 1, 9};
Arrays.sort(arr); // 오름차순

// 내림차순
Integer[] arr2 = {5, 2, 8, 1, 9};
Arrays.sort(arr2, Collections.reverseOrder());

// 버블 정렬
for (int i = 0; i < arr.length - 1; i++) {
    for (int j = 0; j < arr.length - 1 - i; j++) {
        if (arr[j] > arr[j + 1]) {
            int temp = arr[j];
            arr[j] = arr[j + 1];
            arr[j + 1] = temp;
        }
    }
}

// 선택 정렬
for (int i = 0; i < arr.length - 1; i++) {
    int minIdx = i;
    for (int j = i + 1; j < arr.length; j++) {
        if (arr[j] < arr[minIdx]) {
            minIdx = j;
        }
    }
    int temp = arr[i];
    arr[i] = arr[minIdx];
    arr[minIdx] = temp;
}

3.2 탐색

// 선형 탐색
public int linearSearch(int[] arr, int target) {
    for (int i = 0; i < arr.length; i++) {
        if (arr[i] == target) {
            return i;
        }
    }
    return -1;
}

// 이진 탐색 (정렬된 배열)
public int binarySearch(int[] arr, int target) {
    int left = 0;
    int right = arr.length - 1;
    
    while (left <= right) {
        int mid = (left + right) / 2;
        if (arr[mid] == target) {
            return mid;
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    return -1;
}

3.3 재귀

// 팩토리얼
public int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

// 피보나치
public int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

3.4 동적 프로그래밍 (DP)

// 피보나치 (DP)
public int fibonacciDP(int n) {
    int[] dp = new int[n + 1];
    dp[0] = 0;
    dp[1] = 1;
    
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

4. 네트워크

4.1 OSI 7계층

  1. 물리계층 (Physical): 비트 전송
  2. 데이터링크계층 (Data Link): 프레임 전송, 오류 검출
  3. 네트워크계층 (Network): 패킷 라우팅 (IP)
  4. 전송계층 (Transport): 세그먼트 전송 (TCP/UDP)
  5. 세션계층 (Session): 세션 관리
  6. 표현계층 (Presentation): 데이터 변환, 암호화
  7. 응용계층 (Application): 사용자 인터페이스 (HTTP, FTP)

4.2 TCP/IP

  • TCP: 연결 지향, 신뢰성 보장, 속도 느림
  • UDP: 비연결, 신뢰성 낮음, 속도 빠름

4.3 주요 프로토콜

  • HTTP/HTTPS: 웹 통신
  • FTP: 파일 전송
  • SMTP: 이메일 전송
  • DNS: 도메인 이름 해석

5. 운영체제

5.1 프로세스와 스레드

  • 프로세스: 실행 중인 프로그램, 독립적 메모리 공간
  • 스레드: 프로세스 내 실행 단위, 공유 메모리

5.2 스케줄링

  • 선입선출 (FCFS): 먼저 온 것 먼저 처리
  • 라운드 로빈 (Round Robin): 시간 할당량 기반
  • 우선순위: 우선순위가 높은 것 먼저

5.3 동기화

  • 세마포어: 공유 자원 접근 제어
  • 뮤텍스: 상호 배제
